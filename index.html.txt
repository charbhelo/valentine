  <script>
    const PASS = "cartouna";
    const LOVE_TEXT = "Happy Valentine‚Äôs my love, all of me loves all of you ‚ù§Ô∏è";
    const TYPE_SPEED_MS = 45;

    const lock = document.getElementById("lock");
    const pwd = document.getElementById("pwd");
    const openBtn = document.getElementById("openBtn");
    const err = document.getElementById("err");

    const photo = document.getElementById("photo");
    const messageWrap = document.getElementById("messageWrap");
    const message = document.getElementById("message");

    const song = document.getElementById("song");
    const musicWrap = document.getElementById("musicWrap");
    const musicBtn = document.getElementById("musicBtn");

    function typeMessage(text, speed) {
      message.innerHTML = "";
      const cursor = document.createElement("span");
      cursor.className = "cursor";
      cursor.textContent = "|";

      let i = 0;
      const timer = setInterval(() => {
        message.textContent = text.slice(0, i + 1);
        message.appendChild(cursor);
        i++;
        if (i >= text.length) {
          clearInterval(timer);
          setTimeout(() => { cursor.style.display = "none"; }, 800);
        }
      }, speed);
    }

    async function tryPlay() {
      try {
        await song.play();
        musicBtn.textContent = "Music playing ‚ù§Ô∏è";
      } catch {
        musicBtn.textContent = "Play music üéµ";
      }
    }

    function showMain() {
      lock.style.display = "none";
      photo.style.display = "block";
      messageWrap.style.display = "block";
      musicWrap.style.display = "block";
      typeMessage(LOVE_TEXT, TYPE_SPEED_MS);
      tryPlay();
    }

    openBtn.addEventListener("click", () => {
      if (pwd.value === PASS) {
        err.textContent = "";
        showMain();
      } else {
        err.textContent = "Wrong password, try again.";
      }
    });

    pwd.addEventListener("keydown", (e) => {
      if (e.key === "Enter") openBtn.click();
    });

    musicBtn.addEventListener("click", async () => {
      try {
        await song.play();
        musicBtn.textContent = "Music playing ‚ù§Ô∏è";
      } catch {
        musicBtn.textContent = "Tap again";
      }
    });
